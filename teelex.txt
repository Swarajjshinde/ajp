    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {                                         

     try
     {
         Connect c=new Connect();
         con=c.dbconnect();
         String str="insert into student values(?,?,?,?)";
         PreparedStatement pstm=con.prepareStatement(str);
         pstm.setString(1,jroll.getText());
          pstm.setString(2,jname.getText());
          pstm.setString(3,jemail.getText());
          pstm.setString(4,jmobile.getText());
          pstm.executeUpdate();
          JOptionPane.showMessageDialog(this,"Data Insertion Success");
          jroll.setText("");
          jname.setText("");
          jmobile.setText("");
          jemail.setText("");
     }
          catch(SQLException ex)
                  {
                  Logger.getLogger(TEELEX.class.getName()).log(Level.SEVERE,null,ex);
                  }
        
    }
